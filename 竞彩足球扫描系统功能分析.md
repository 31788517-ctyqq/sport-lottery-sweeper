### 1.1 架构概览

### 1.2 核心架构原则

**关注点分离**：数据采集、数据处理、数据存储、前端展示相互解耦，便于独立扩展和维护。

**可扩展性**：
- **水平扩展**：爬虫Worker节点可根据数据源负载自动扩缩容
- **垂直扩展**：支持按优先级动态分配计算资源（高优先级情报使用更多资源）

**高可用性**：
- 多副本部署（关键服务至少3个副本）
- 熔断降级机制（爬虫服务异常时自动切换至备用数据源）
- 分布式锁（防止重复抓取）

**可观测性**：
- 结构化日志（ELK Stack）
- 分布式追踪（OpenTelemetry）
- 业务指标监控（数据覆盖率、情报时效性）

### 2.2 智能调度系统

#### **动态抓取策略算法**

```python
class IntelligentScheduler:
    def __init__(self):
        self.priority_queue = PriorityQueue()
        self反爬策略 = {
            'aggressive': 30,  # 激进策略：30秒/次
            'normal': 60,      # 正常策略：60秒/次
            'conservative': 300  # 保守策略：5分钟/次
        }
    
    def schedule_crawl(self, data_source):
        """根据数据源重要性、反爬强度、历史成功率动态调整"""
        importance_score = self.calculate_importance(data_source)
        反爬强度 = self.assess_anti_crawl_difficulty(data_source)
        历史成功率 = self.get_historical_success_rate(data_source)
        
        # 动态计算抓取频率
        base_interval = 60  # 基础间隔1分钟
        importance_factor = importance_score / 10
        反爬惩罚 = 反爬强度 / 10
        成功率奖励 = 历史成功率 / 10
        
        interval = base_interval * 反爬惩罚 / (importance_factor + 成功率奖励)
        
        # 添加到优先级队列
        self.priority_queue.push(
            task=CrawlTask(data_source, interval),
            priority=importance_score
        )
    
    def handle_blockage(self, data_source):
        """IP被封时的自动降级策略"""
        if data_source.block_count >= 3:
            # 自动切换到保守策略
            self.反爬策略[data_source] = 'conservative'
            # 切换代理IP池
            data_source.switch_proxy_pool()
            # 记录降级事件
            self.log_incident(data_source, '降级至保守策略')
```

```python
class AdaptiveDegradation:
    def __init__(self):
        self.failure_count = {}
        self.degradation_level = {}
    
    def check_and_degrade(self, data_source):
        if self.failure_count[data_source] >= 3:
            # 触发降级
            if self.degradation_level[data_source] < 2:
                self.degradation_level[data_source] += 1
                self.log_degradation(data_source)
            
            # 根据降级级别调整策略
            if self.degradation_level[data_source] == 1:
                # 第一级降级：增加请求间隔
                return self.反爬策略['conservative']
            elif self.degradation_level[data_source] == 2:
                # 第二级降级：切换代理池
                data_source.switch_proxy_pool()
                return self.反爬策略['conservative']
            else:
                # 第三级降级：暂停抓取
```

```javascript
class DataManager {
    constructor() {
        this.matches = [];
        this.intelligence = {};
        this.predictions = {};
    }
    
    // 初始化加载数据
    async initialize() {
        await this.loadLeagues();

    }
    
    // 加载联赛列表
    async loadLeagues() {
        try {
            const response = await fetch('/api/leagues');
            const leagues = await response.json();
            this.renderLeagues(leagues);
        } catch (error) {
            console.error('加载联赛失败:', error);
            this.showError('加载联赛列表失败');
        }
    }
    
    // 加载比赛数据
    async loadMatches() {
        try {
            this.showLoading();
            const response = await fetch('/api/matches');
            this.matches = await response.json();
            
            // 按日期分组
            const grouped = this.groupMatchesByDate(this.matches);
            this.renderMatches(grouped);
            this.updateLastUpdateTime();
        } catch (error) {
            console.error('加载比赛失败:', error);
            this.showError('加载比赛数据失败');
        } finally {
            this.hideLoading();
        }
    }
    
    // 加载比赛情报
    async loadMatchIntelligence(matchId) {
        try {
            const [intelResponse, predResponse] = await Promise.all([
                fetch(`/api/intelligence/${matchId}`),
                fetch(`/api/predictions/${matchId}`)
            ]);
            
            const intelligenceData = await intelResponse.json();
            const predictionsData = await predResponse.json();
            
            this.intelligence[matchId] = {
                intelligence: intelligenceData,
                predictions: predictionsData
            };
            
            this.renderMatchIntelligence(matchId);
        } catch (error) {
            console.error(`加载比赛${matchId}情报失败:`, error);
        }
    }
    
    // 自动刷新机制
    setupAutoRefresh() {
        setInterval(async () => {
            await this.loadMatches();
        }, 1800000);  // 30分钟
    }
}
```

**2. 筛选与排序系统**
```javascript
class FilterManager {
    constructor() {
        this.currentLeague = 'all';
        this.currentCategory = 'all';
        this.currentSource = 'all';
        this.currentSort = 'time';
    }
    
    // 应用所有筛选条件
    applyFilters() {
        // 先筛选比赛卡片
        this.filterMatchCards();
        
        // 再筛选情报项
        this.filterIntelligenceItems();
        
        // 最后应用排序
        this.applySorting();
        
        // 更新空状态提示
        this.updateEmptyState();
    }
    
    // 筛选比赛卡片
    filterMatchCards() {
        document.querySelectorAll('.match-card').forEach(card => {
            const league = card.getAttribute('data-league');
            const isVisible = this.currentLeague === 'all' || league === this.currentLeague;
            card.classList.toggle('hidden-card', !isVisible);
        });
    }
    
    // 筛选情报项
    filterIntelligenceItems() {
        document.querySelectorAll('.news-item').forEach(item => {
            const category = item.getAttribute('data-type');
            const source = item.getAttribute('data-source');
            const parentCard = item.closest('.match-card');
            const isCardVisible = !parentCard.classList.contains('hidden-card');
            
            const isCategoryMatch = this.currentCategory === 'all' || category === this.currentCategory;
            const isSourceMatch = this.currentSource === 'all' || source === this.currentSource;
            
            item.classList.toggle('hidden-news', !(isCategoryMatch && isSourceMatch && isCardVisible));
        });
    }
    
    // 应用排序
    applySorting() {
        document.querySelectorAll('.match-body').forEach(body => {
            const items = Array.from(body.querySelectorAll('.news-item:not(.hidden-news)'));
            
            if (this.currentSort === 'weight') {
                // 按权重排序（降序）
                items.sort((a, b) => {
                    const wA = parseFloat(a.getAttribute('data-weight')) || 0;
                    const wB = parseFloat(b.getAttribute('data-weight')) || 0;
                    return wB - wA;
                });
            } else {
                // 按时间排序（降序）
                items.sort((a, b) => {
                    const tA = new Date(a.getAttribute('data-time') || 0);
                    const tB = new Date(b.getAttribute('data-time') || 0);
                    return tB - tA;
                });
            }
            
            // 重新排列DOM
            items.forEach(item => body.appendChild(item));
        });
    }
}
```

2. **智能降级机制**：检测到IP封锁自动切换至保守策略
3. **指纹伪装**：修改Canvas、WebGL、字体栈等浏览器指纹

5. **持续更新**：建立反爬策略库，定期更新User-Agent和请求头

### 5.2 数据去重与融合

**难点**：同一情报可能来自多个来源，需要智能去重并融合

**解决方案**：
```python
class DataDeduplication:
    def __init__(self):
        self.similarity_threshold = 0.85
    
    def deduplicate_intelligence(self, intelligence_list):
        """智能去重情报"""
        unique_intelligence = []
        
        for intel in intelligence_list:
            is_duplicate = False
            
            for unique in unique_intelligence:
                # 计算相似度
                similarity = self.calculate_similarity(intel, unique)
                
                if similarity > self.similarity_threshold:
                    # 融合数据（保留更权威的来源）
                    if self.is_more_authoritative(intel.source, unique.source):
                        unique_intelligence.remove(unique)
                        unique_intelligence.append(intel)
                    is_duplicate = True
                    break
            
            if not is_duplicate:
                unique_intelligence.append(intel)
        
        return unique_intelligence
    
    def calculate_similarity(self, intel1, intel2):
        """计算两条情报的相似度"""
        # 使用编辑距离计算内容相似度
        content_similarity = self.levenshtein_ratio(
            intel1.content, 
            intel2.content
        )
        
        # 时间相似度（1小时内为1分）
        time_diff = abs(
            (intel1.publish_time - intel2.publish_time).total_seconds()
        )
        time_similarity = 1 if time_diff < 3600 else 0
        
        # 综合相似度
        return (content_similarity * 0.7) + (time_similarity * 0.3)
```

### 5.3 实时数据推送

**难点**：前端需要实时更新数据，避免频繁刷新页面

**解决方案**：
```javascript
// WebSocket实时推送
class RealTimeUpdater {
    constructor() {
        this.ws = null;
        this.reconnectInterval = 5000;
        this.isConnected = false;
    }
    
    connect() {
        this.ws = new WebSocket('wss://api.example.com/ws');
        
        this.ws.onopen = () => {
            console.log('WebSocket连接已建立');
            this.isConnected = true;
            this.subscribeToUpdates();
        };
        
        this.ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.handleUpdate(data);
        };
        
        this.ws.onclose = () => {
            console.log('WebSocket连接已断开');
            this.isConnected = false;
            this.scheduleReconnect();
        };
    }
    
    subscribeToUpdates() {
        // 订阅比赛情报更新
        this.ws.send(JSON.stringify({
            type: 'subscribe',
            channel: 'intelligence',
            filters: {
                league: this.currentLeague,
                category: this.currentCategory
            }
        }));
        
        // 订阅系统状态更新
        this.ws.send(JSON.stringify({
            type: 'subscribe',
            channel: 'system_status'
        }));
    }
    
    handleUpdate(data) {
        switch (data.type) {
            case 'new_intelligence':
                this.addIntelligence(data.payload);
                this.showNotification('有新的情报更新');
                break;
            case 'system_status':
                this.updateSystemStatus(data.payload);
                break;
        }
    }
    
    scheduleReconnect() {
        setTimeout(() => this.connect(), this.reconnectInterval);
    }
}
```
